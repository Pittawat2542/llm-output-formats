

# Example
```xml
<?xml version="1.0"?>
<game>
	<id>1</id>
	<title>The Quest</title>
	<objective>To find the lost treasure</objective>
	<description>This is a quest to find the lost treasure</description>
	<reward>A magic sword</reward>
	<quest-giver>The King</quest-giver>
	<tasks>
		<task>
			<order>1</order>
			<objective>Find the lost treasure</objective>
			<description>Find the lost treasure</description>
			<location>The Lost Treasure</location>
		</task>
		<task>
			<order>2</order>
			<objective>Find the lost treasure</objective>
			<description>Find the lost treasure</description>
			<location>The Lost Treasure</location>
		</task>
	</tasks>
</game>
```
