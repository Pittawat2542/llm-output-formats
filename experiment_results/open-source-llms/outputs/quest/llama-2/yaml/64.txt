```yaml
  objective: "Complete the quest."
  quest_giver: "The quest giver."
  reward: "A reward."
  tasks:

    location: "The location."
    objective: "Complete the task."
    order: 1
  - description: "Another task."
    location: "The location."
    objective: "Complete the task."
    order: 2
  title: "The quest title."
```
