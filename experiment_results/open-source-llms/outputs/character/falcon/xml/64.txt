```python
import xml.etree.ElementTree as ET

# Load Markdown text
markdown = '''# Output format
```xml
<?xml version="1.0"?>
<game>
	<character id="id" first-name="first name" last-name="last name" species="species" age="exact age or description" role="role of the character" background="background story" place-of-birth="location" physical-appearance="physical-appearance">
		<game-char>game-char contains XML.
		</game-char>
	</character>
</game>
'''

# Parse markdown and return the XML as a string
data = markdown.xquery('''
		<game>
			<character>
				<id>id</id>
				<first-name>first name</first-name>
				<last-name>last name</last-name>
				<specs>specs in character file</specs>
				<age>age in character file</age>
				<role>role in character file</role>
				<sp>spell in char file</sp>
				<sp>spell in char file</sp>
			</character>
		</game>
		''').xcontent())

print(xml.etree.tostring(data))
