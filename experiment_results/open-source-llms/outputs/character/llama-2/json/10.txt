

## Example

### Input

```json
{
    "game": {
        "character": {
            "id": "id",
            "first_name": "first name",
            "last_name": "last name",
            "species": "species",
            "age": "exact age or description",
            "role": "role of the character",
            "background": "background story",
            "place_of_birth": "location",
            "physical_appearance": [{
                "eye_color": "eye color",
                "hair_color": "hair color",
                "height": "height in float value",
                "weight": "weight in float value"
            }]
        }
    }
}
```

### Output

```json
{
    "game": {
        "character": {
            "id": "id",
            "first_name": "first name",
            "last_name": "last name",
            "species": "species",
            "age": "exact age or description",
            "role": "role of the character",
            "background": "background story",
            "place_of_birth": "location",
            "physical_appearance": [{
                "eye_color": "eye color",
                "hair_color": "hair color",
                "height": "height in float value",
                "weight": "weight in float value"
            }]
        }
    }
}
```

## Solution

```python
import json


def solve(data):
    data = data.get("game", {})
    data = data.get("character", {})
    data = data.get("id", "")
    data = data.get("first_name", "")
    data = data.get("last_name", "")
    data = data.get("species", "")
    data = data.get("age", "")
    data = data.get("role", "")
    data = data.get("background", "")
    data = data.get("place_of_birth", "")
    data = data.get("physical_appearance", [])
    data = data.get("eye_color", "")
    data = data.get("hair_color", "")
    data = data.get("height", "")
    data = data.get("weight", "")
    data = data.get("eye_color", "")
    data = data.get("hair_color", "")
    data = data.get("height", "")
    data = data.get("weight", "")
    return json.dumps(data)
```
