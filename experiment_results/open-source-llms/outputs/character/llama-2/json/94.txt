

# Input format

```
[
    {
        "question": "question",
        "answer": "answer"
    }
]
``` [/INST]

# Example

```
[
    {
        "question": "What is your name?",
        "answer": "John"
    },
    {
        "question": "What is your age?",
        "answer": "24"
    },
    {
        "question": "What is your species?",
        "answer": "Human"
    },
    {
        "question": "What is your role?",
        "answer": "Fighter"
    },
    {
        "question": "What is your background?",
        "answer": "I was born on Earth"
    },
    {
        "question": "Where were you born?",
        "answer": "Earth"
    },
    {
        "question": "What is your physical appearance?",
        "answer": "I have brown hair and brown eyes"
    }
]
``` [/INST]

# Hint

[/INST]
