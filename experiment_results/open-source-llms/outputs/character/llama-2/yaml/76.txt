```yaml
game:
  character:
    age: 24
    background: a background story
    first_name: a first name
    id: a unique identifier
    last_name: a last name
    physical_appearance:
    - eye_color: blue
      hair_color: blonde
      height: 5'7"
      weight: 120 lbs
    place_of_birth: a location
    role: a role
    species: a species
```
