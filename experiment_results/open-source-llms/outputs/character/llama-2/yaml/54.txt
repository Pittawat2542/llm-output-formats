

### Example
```yaml
game:
  character:
    age: 21
    background: "I'm a student at the University of Toronto"
    first_name: "Jane"
    id: "123456789"
    last_name: "Doe"
    physical_appearance:
    - eye_color: blue
      hair_color: brown
      height: 160
      weight: 50
    place_of_birth: "Toronto, Canada"
    role: "Student"
    species: "Human"
```
