

## Examples

### Example 1
```xml
<?xml version="1.0"?>
<game>
	<character>
		<id>1</id>
		<first-name>John</first-name>
		<last-name>Doe</last-name>
		<species>Human</species>
		<age>32</age>
		<role>Hero</role>
		<background>John is a hero who is trying to save the world from evil.</background>
		<place-of-birth>New York</place-of-birth>
		<physical-appearance>
			<physical-appearance>
				<eye-color>Blue</eye-color>
				<hair-color>Blonde</hair-color>
				<height>180</height>
				<weight>80</weight>
			</physical-appearance>
		</physical-appearance>
	</character>
</game>
```
