

## Examples

### Example 1

```
[INST] <<SYS>>

# Example 1

game:
  enemy:
    id: 1
    name: "Enemy 1"
    description: "This is the description of Enemy 1."
    stats:
      hp: 100
      mp: 100
      atk: 100
      def: 100
      spd: 100

# Output format
{
    "game": {
        "enemy": {
            "id": "1",
            "name": "Enemy 1",
            "description": "This is the description of Enemy 1.",
            "stats": {
                "hp": "100",
                "mp": "100",
                "atk": "100",
                "def": "100",
                "spd": "100"
            }
        }
    }
}

[/SYS]
```
