

# Example

```xml
<?xml version="1.0"?>
<game>
	<enemy>
		<id>123456789</id>
		<name>Boss</name>
		<description>Boss</description>
		<stats>
			<hp>1000</hp>
			<mp>1000</mp>
			<atk>1000</atk>
			<def>1000</def>
			<spd>1000</spd>
		</stats>
	</enemy>
</game>
```
