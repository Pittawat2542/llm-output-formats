

### Example
```xml
<?xml version="1.0"?>
<game>
	<enemy>
		<id>1</id>
		<name>King of the Fairies</name>
		<description>A powerful fairy who lives in the forest. He is the king of all the fairies.</description>
		<stats>
			<hp>100</hp>
			<mp>100</mp>
			<atk>100</atk>
			<def>100</def>
			<spd>100</spd>
		</stats>
	</enemy>
</game>
```
