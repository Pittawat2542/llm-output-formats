

# Example
```xml
<?xml version="1.0"?>
<game>
	<enemy>
		<id>1</id>
		<name>Bob</name>
		<description>A nice guy</description>
		<stats>
			<hp>100</hp>
			<mp>100</mp>
			<atk>100</atk>
			<def>100</def>
			<spd>100</spd>
		</stats>
	</enemy>
</game>
```
