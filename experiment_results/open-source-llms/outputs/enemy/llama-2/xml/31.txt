

### Example
```xml
<?xml version="1.0"?>
<game>
	<enemy>
		<id>id</id>
		<name>name</name>
		<description>description</description>
		<stats>
			<hp>100</hp>
			<mp>100</mp>
			<atk>100</atk>
			<def>100</def>
			<spd>100</spd>
		</stats>
	</enemy>
</game>
```

### Input format
```xml
<?xml version="1.0"?>
<game>
	<enemy>
		<id>id</id>
		<name>name</name>
		<description>description</description>
		<stats>
			<hp>hp int value</hp>
			<mp>mp int value</mp>
			<atk>atk int value</atk>
			<def>def int value</def>
			<spd>spd int value</spd>
		</stats>
	</enemy>
</game>
```
