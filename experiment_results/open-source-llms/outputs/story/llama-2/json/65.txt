import json


def generate_game_story(questions):
    game = {
        "title": questions[0]["answers"][0],
        "synopsis": questions[1]["answers"][0],
        "beginning": questions[2]["answers"][0],
        "ending": questions[3]["answers"][0],
    }
    return json.dumps(game)
