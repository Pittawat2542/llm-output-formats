Here is a Python script that generates an enemy battle status in the specified YAML format:

```python
import yaml
import random

def generate_enemy():
    enemy_id = str(random.randint(1000, 9999))
    enemy_name = f"Enemy {enemy_id}"
    description = "A fearsome foe stands before you."
    
    atk = random.randint(10, 30)
    def_value = random.randint(5, 20)
    hp = random.randint(50, 200)
    mp = random.randint(10, 100)
    spd = random.randint(5, 15)

    enemy_status = {
        "game": {
            "enemy": {
                "description": description,
                "id": enemy_id,
                "name": enemy_name,
                "stats": {
                    "atk": atk,
                    "def": def_value,
                    "hp": hp,
                    "mp": mp,
                    "spd": spd
                }
            }
        }
    }

    return yaml.dump(enemy_status, default_flow_style=False)

print(generate_enemy())
```

This script will generate a random enemy with stats within the specified ranges and output it in YAML format. The `yaml.dump()` function is used to convert the Python dictionary into YAML format.
