Here's a Python code to generate an enemy battle status and output it in YAML format.

```python
import yaml
import random

def generate_enemy_status():
    descriptions = [
        "A fearsome dragon with scales as black as coal.",
        "An undead warrior risen from the depths of hell.",
        "A giant spider with eyes that glow like lanterns."
    ]

    enemy_status = {
        "game": {
            "enemy": {
                "description": random.choice(descriptions),
                "id": str(random.randint(1, 100)),
                "name": f"Enemy {random.randint(1, 10)}",
                "stats": {
                    "atk": random.randint(10, 50),
                    "def": random.randint(5, 20),
                    "hp": random.randint(20, 100),
                    "mp": random.randint(0, 20),
                    "spd": random.randint(1, 10)
                }
            }
        }
    }

    return yaml.dump(enemy_status, default_flow_style=False)

print(generate_enemy_status())
```

The output will look something like this:

```yaml
game:
  enemy:
    description: An undead warrior risen from the depths of hell.
    id: '14'
    name: Enemy 8
    stats:
      atk: 31
      def: 16
      hp: 67
      mp: 11
      spd: 7
```
