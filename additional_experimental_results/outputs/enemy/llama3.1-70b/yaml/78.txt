Here is a simple Python function that generates an enemy battle status in YAML format.

```python
import yaml
import random

def generate_enemy_status():
    # Define the enemy stats range
    min_stat = 10
    max_stat = 100

    # Generate random stats for the enemy
    atk = random.randint(min_stat, max_stat)
    def_val = random.randint(min_stat, max_stat)
    hp = random.randint(min_stat * 2, max_stat * 2)  # HP tends to be higher than other stats
    mp = random.randint(min_stat // 2, max_stat // 2)  # MP tends to be lower than other stats
    spd = random.randint(min_stat, max_stat)

    # Define a list of possible enemy names and descriptions
    enemy_names = ["Orc", "Troll", "Goblin", "Dragon"]
    enemy_descriptions = [
        "A fierce warrior from the mountains.",
        "A hulking beast from the forest.",
        "A cunning foe from the caves.",
        "A legendary creature of fire and destruction."
    ]

    # Select a random name and description
    name = random.choice(enemy_names)
    description = random.choice(enemy_descriptions)

    # Generate a unique ID (in this case, a random integer)
    id = random.randint(1, 1000)

    # Create the enemy status YAML output
    output = {
        "game": {
            "enemy": {
                "description": description,
                "id": id,
                "name": name,
                "stats": {
                    "atk": atk,
                    "def": def_val,
                    "hp": hp,
                    "mp": mp,
                    "spd": spd
                }
            }
        }
    }

    return yaml.dump(output, default_flow_style=False)


# Usage example:
print(generate_enemy_status())
```

Please note that the actual output will vary because it's randomly generated.
