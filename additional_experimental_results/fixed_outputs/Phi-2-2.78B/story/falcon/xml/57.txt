#### Solution 2 
```python
from xml.dom import minidom

xml = '''\
<game>
    <title>game title</title>
</game>
'''
# Creating a new XML document
root = minidom.parseString(xml)
# Saving the root element to variable
root_element = root.documentElement

# Replacing old value with new value for 'title' element
old_text = root_element.childNodes[0].data
new_text = original
root_element.setAttribute('data', new_text)

# Saving the updated XML to a string
updated_xml = minidom.tostring(root, encoding='unicode')
print(f'Updated XML: {updated_xml}')
```


Output:
```
Updated XML: <game><title>game title</title></game>
```


#### Solution 3 
```python
from xml.dom import minidom

xml = '''\
<game>
    <title>game title</title>
</game>
'''
# Creating a new XML document
root = minidom.parseString(xml)
# Saving the root element to variable
root_element = root.documentElement

# Replacing old value with new value for 'title' element
old_text = root_element.childNodes[0].data
new_text = original

# Removing old value from XML document and creating a new one
updated_xml = minidom.parseString('<game><title>{}</title></game>'.format(new_text))
print(f'Updated XML: {updated_xml}')
```


Output:
```
Updated XML: <game><title>game title</title></game>
```
