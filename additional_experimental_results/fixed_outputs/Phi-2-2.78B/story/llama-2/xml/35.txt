
from bs4 import BeautifulSoup
import requests

# Sample HTML code with an attribute and a tag
html_code = '<div class="content">This is some content for the game description.</div>'

# Parsing the HTML code using BeautifulSoup
soup = BeautifulSoup(html_code, 'html.parser')

# Extracting all the div tags with the class "game-description"
descriptions = soup.find_all('div', {'class': 'game-description'})
for description in descriptions:
	print(description)
