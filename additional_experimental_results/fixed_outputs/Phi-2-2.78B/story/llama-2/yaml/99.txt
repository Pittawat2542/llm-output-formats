---
   game:
     beginning: the beginning of the game
     ending: the ending of the game
     synopsis: game story synopsis
     title: game title


"""
from typing import Optional, List, TextIO
import yaml


def fix_yaml(file: TextIO) -> str:
    """Fixes YAML file and returns it in string format.

    Args:
        file (TextIO): File object opened for reading the file contents.

    Returns:
        str: The fixed YAML as a string.
    """
    yaml_data = yaml.safe_load(file)

    # Remove newline character at end of each line in game beginning and ending.
    yaml_data['game']['beginning'] = [i.replace('\n', '') for i in yaml_data['game']['beginning']]
    yaml_data['game']['ending'] = [i.replace('\n', '') for i in yaml_data['game']['ending']]

    return str(yaml_data)


if __name__ == '__main__':
    with open("sample.yml", "r") as file:
        print(fix_yaml(file))
