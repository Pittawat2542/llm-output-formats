import os
from yaml import load, dump

filename = 'enemy.yaml'  # File path to the YAML data
filepath = os.path.join(os.getcwd(), filename)

with open(filepath, 'r') as stream:
    data = load(stream)
    print(f"Original: {dump(data, default_flow_style=False)}")

    for key, value in data['enemy'].items():
        if value is None:
            del data['enemy'][key]

    with open(filepath, 'w') as stream:
        dump(data, stream)
