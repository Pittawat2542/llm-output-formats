```python
"""
# Python 3 program to demonstrate 
# of how to write XML document using xmltodict
import xmltodict
import xml.dom.minidom

# Creating sample data
data = {'game':{'character':{'id': '1', 'first-name': 'John', 'last-name': 'Doe', 
                                 'species': 'human', 'age': '30', 'role': 'warrior', 
                                 'background': 'He was born in a small village.', 
                                 'place-of-birth': 'village', 
                                 'physical-appearance':{'eye-color':'blue','hair-color':'brown','height':1.7,'weight':70}}}

# Converting the sample data to XML document using xmltodict function
xml_data = xml.dom.minidom.parseString(xmltodict.unparse(data))
print("XML Data: ", xml_data)
```
