from typing import List, Dict

def fix_grammar(data: Dict[str, str]) -> Dict[str, str]:
    """
    Corrects grammar in a YAML file.

    Args:
        data (Dict[str, str]): Dictionary containing the game data.

    Returns:
        Dict[str, str]: Corrected game data with correct grammar.
    """
    if "dialogues" not in data.keys():
        return data
    for dialogue in data["dialogues"]:
        if "order" not in dialogue.keys() or "speaker" not in dialogue.keys() or "text" not in dialogue.keys():
            continue
        if dialogue["order"] == "dialogue order":
            del dialogue["order"]
        if type(dialogue["speaker"]) is str:
            dialogue["speaker"] = [dialogue["speaker"]]
    return data
