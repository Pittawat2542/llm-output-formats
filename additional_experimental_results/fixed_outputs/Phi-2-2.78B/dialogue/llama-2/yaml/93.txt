from typing import List
import random

def generate_dialogue(speaker: str, options: List[str], length: int) -> str:
    """
    Generates a random dialogue for the given speaker with the specified options and length.

    Args:
        speaker (str): The name of the speaker.
        options (List[str]): A list of possible dialogue options to choose from.
        length (int): The desired length of the generated dialogue in number of characters.

    Returns:
        A randomly generated dialogue for the given speaker with the specified options and length.
    """
    dialogue = ""
    for i in range(length // 2):
        option = random.choice(options)
        if i % 2 == 0:
            dialogue += f"{speaker}: {option}\n"
        else:
            dialogue += f"You: {option}\n"
    return dialogue
