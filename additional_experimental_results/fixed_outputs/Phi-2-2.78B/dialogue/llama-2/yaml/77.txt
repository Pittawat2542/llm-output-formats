from typing import List, Tuple
import random
import string

# Function to generate a random password of length n using alphanumeric characters.
def generate_password(n) -> str:

    return ''.join(random.choices(string.ascii_letters + string.digits, k=n))

# Function to generate a list of random strings of length m using the above function for each string in the list.
def generate_strings(m, num) -> List[str]:

    return [generate_password(m) for _ in range(num)]

print(generate_strings(10, 5))
