```python
from typing import List, Dict
import re


def extract_dialogue(lines: List[str]) -> List[Dict]:
    """
    Extracts the dialogue from a given list of lines.
    A line is considered to be part of dialogue if it starts with '>'.

    Parameters:
        - lines (List): A list of strings, each containing a single line of text.

    Returns:
        - dialogues (List[Dict]): A list of dictionaries representing the extracted
                                 dialogues, where each dictionary has the keys
                                 "speaker" and "text".

    Examples:
        >>> lines = ["> Player 1: Hello!", "Player 2: Hi there.", "> Player 1: How are you?",
                     "> Player 2: I'm doing well, thanks for asking."]
        >>> extract_dialogue(lines)
        [{"speaker": "Player 1", "text": "Hello!"}, {"speaker": "Player 2", "text": "Hi there."}]

    """
    # Extract the lines starting with '>' as dialogues
    dialogues = []
    for line in lines:
        if line.startswith(">"):
            # Remove any leading whitespace
            line = line.lstrip()
            # Split the dialogue into speaker and text
            speaker, text = re.split("\s+", line)
            dialogue = {"speaker": speaker, "text": text}
            dialogues.append(dialogue)

    return dialogues


lines = ["> Player 1: Hello!", "Player 2: Hi there.", "> Player 1: How are you?",
         "Player 2: I'm doing well, thanks for asking."]
dialogues = extract_dialogue(lines)
print(dialogues) # Output: [{"speaker": "Player 1", "text": "Hello!"}, {"speaker": "Player 2", "text": "Hi there."}]
```
