```python
from bs4 import BeautifulSoup 
import requests 

# Send request to the website
r = requests.get("https://en.wikipedia.org/wiki/Lists_of_the_largest_companies")
html_content = r.text

# Create an instance of soup
soup = BeautifulSoup(html_content, "lxml")

# Find the game element 
game_elem = soup.find('game')

if game_elem == None: 
	print("Error: no game element found!")
else:

	# Remove whitespace from the XML tags 
	for tag in game_elem.strings: 
		tag.strip()

	# Replace <i> with </i> for easier reading 
	for tag in game_elem.findAll('i'): 
		tag.replaceWith('</i><i>')

	# Fix key completeness error 
	if 'quest-giver' not in str(game_elem): 
		new_key = "quest giver"
		game_elem['quest-giver']=new_key
	print(str(game_elem))
```
