import xml.etree.ElementTree as ET

# Parse the XML string
xml_str = '''
<game>
    <id>1</id>
	<title>The quest title</title>
	<objective>The quest objective</objective>
	<description>The quest description</description>
	<reward>The quest reward</reward>
	<quest-giver>The quest giver</quest-giver>
	<tasks>
		<task>
			<order>1</order>
			<objective>The task objective</objective>
			<description>The task description</description>
			<location>The task location</location>
		</task>
	</tasks>
</game>
'''
root = ET.fromstring(xml_str)
print(ET.tostring(root))
