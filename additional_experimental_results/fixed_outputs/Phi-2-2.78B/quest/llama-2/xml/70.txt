# Question: What is the corrected syntax in the XML?

```python
from typing import List, Tuple
import re
from xml.etree import ElementTree as ET

def correct_xml(xml_data: str) -> str:

    # Split input data into lines and remove whitespace
    lines = list(map(lambda x: x.strip(), xml_data.split('\n'))) 

    # Create a pattern to match invalid characters
    pattern = re.compile(r'[^a-zA-Z0-9 \t\.\!\-]')

    # Initialize variable for new data
    new_lines: List[str] = []

    for line in lines:
        
        # Remove invalid characters from the line
        line = pattern.sub('', line)

        # Add space before and after a semicolon or comma
        line = re.sub(r';', ' ; ', line)
        line = re.sub(r',' , ', line)

        # Split lines into list of words
        words = line.split() 

        # Remove any empty words from the list
        words = list(filter(None, words))

        # Rebuild the line using new words
        new_line: str = ' '.join(words) + '\n'

        # Add the new line to the new data variable
        new_lines.append(new_line) 

    # Create an ElementTree from the new list of lines
    tree: ET.ElementTree = ET.fromstring(''.join(new_lines))

    return tree.getroot().text
```
