from bs4 import BeautifulSoup as bs
import requests


def fix_xml(original):
    original = str(original)
    data = original.replace('\n', ' ').split('\r')[0]
    data = data.strip()

    response = requests.get(data)

    html_text = response.content

    soup = bs(html_text, "lxml")

    game = soup.find("game")

    if not game:
        raise ValueError("Invalid XML")

    id = game.attrs["id"]
    title = game.find("title").string
    objective = game.find("objective").string
    description = game.find("description").string
    reward = game.find("reward").string
    quest_giver = game.find("quest-giver").string
    tasks = []

    for task in game.findAll("task"):
        order = task["order"]
        objective = task["objective"]
        description = task["description"]
        location = task["location"]

        tasks.append(
            {
                "order": order,
                "objective": objective,
                "description": description,
                "location": location,
            }
        )

    return {
        "id": id,
        "title": title,
        "objective": objective,
        "description": description,
        "reward": reward,
        "quest-giver": quest_giver,
        "tasks": tasks,
    }
