import yaml

input_data = r"""game:
  description: quest description
  id: id
  objective: quest objective
  quest_giver: quest giver
  reward: quest reward
  tasks:
  - description: task description
    location: task location
    objective: task objective
    order: task order
  title: quest title"""

with open('input.yaml', 'w') as f:
    f.write(input_data)


def fix_data():
    fixed_game = None
    try:
        data = yaml.safe_load(open('input.yaml'))
        fixed_game = data['game']
    except Exception as e:
        print("There was an error when fixing the file")
        raise e

    if not fixed_game:
        raise TypeError("The game is empty")

    return yaml.dump(fixed_game, default_flow_style=False)

output = fix_data()
