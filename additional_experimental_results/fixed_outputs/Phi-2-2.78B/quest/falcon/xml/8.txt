def fix_xml(data: str):
    """Fixes the syntax of the given XML data.

    Args:
        data (str): The data to be fixed.

    Returns:
        str: Returns a string with the corrected data.
    """
    from re import sub, MULTILINE
    fix_list = [
        # Replace newlines
        ("\n", ""),
        # Replace linebreaks within quotes
        (r"\"(\\.|[^"])*?\"", r'\\1'),
        # Remove tags inside of quote marks
        (r"\".*?\">", ''),
        # Fix incomplete tag
        (r"<([^/]+)/(.*?)>", '<\g<1>\g<2></\g<1>>'),
        # Replace incorrect xml
        ("invalid xml", "invalid_xml")
    ]

    for regex, replacement in fix_list:
        data = sub(regex, replacement, data, flags=MULTILINE)
    return data
