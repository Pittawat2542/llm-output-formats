import xml.etree.ElementTree as ET
# Parse the XML document
tree = ET.parse('quest_for_treasure.xml')
root = tree.getroot()
# Extract the task order
task_order = root.findall('.//task/order[@name="1"]')[0].text

# Extract the objective of each task
for i, task in enumerate(root.iter('task')):

    if order is not None:
