game = """
game:
  description: The Dark Sorcerer's Tower has been taken over by the dark sorcerer, Malyster. He seeks to use its ancient magic to conquer the realm.
  id: QS001
  objective: Defeat Malyster and claim the tower as your own.
  quest_giver: The Wizard of the White Tower
  reward:
    items:
      - 10 gold pieces
      - +2 sword
    experience: 1000 points
  tasks:
  - description: Explore the dark sorcerer'S tower and find the source of its power.
    location: Dark Sorcerer'S Tower, level 5
    objective: Find the ancient tome containing Malyster'S spellbook.
    order: 1
  - description: Infiltrate Malyster's chambers and steal his spellbook.
    location: Dark Sorcerer'S Tower, level 10
    objective: Obtain a copy of Malyster'S spellbook without being detected.
    order: 2
"""

# First step is to split the game into separate lines for easier parsing
game_lines = game.split('\n')

# Next, we can use list comprehension to find all missing keys in each section of the YAML file
missing_keys = [key for line in game_lines for key in ['title', 'description'] if not re.search(r"{}".format(key),line)]

# We then need to loop through each missing key and add it back into the YAML with a default value
for key in missing_keys:
    if key == 'title':
        game = game.replace('title:', key, 1)
    else:
        game = game.replace(key + ':', key, 1)

# Lastly, we can split the game back into separate lines and print it out in YAML format
print(yaml.dump(yaml.safe_load(game), default_flow_style=False))
