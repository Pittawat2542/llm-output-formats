import re
from yaml import load, dump


def fix_yaml(yamlfile):
    data = load(open(yamlfile))
    if 'key_completeness' in data.keys():
        if not any(k in data['quest_giver'] for k in ['title', 'description']):
            data['quest_giver']['title'] = 'Quest Title'
            data['quest_giver']['description'] = 'Quest Description'
    return dump(data, open('fixed_yaml.yml','w'), default_flow_style=False)
