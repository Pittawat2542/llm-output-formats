```xml
<game>
	<enemy>
			<id>id</id>
			<name>name</name>
			<description>description</description>
			<stats>
				<hp>hp int value</hp>
				<mp>mp int value</mp>
				<atk>atk int value</atk>
				<def>def int value</def>
				<spd>spd int value</spd>
			</stats>
		</enemy>
	</game>
</game>
```
