	<?xml version="1.0" ?>
	<script language="javascript">
		function playGame() {
			var xhr = new XMLHttpRequest();
			xhr.open("GET", "game.xml", true);
			xhr.onreadystatechange = function() {
				if (xhr.readyState == 4 && xhr.status == 200) {
					document.getElementById("game").innerHTML = JSON.stringify(xhr.responseText);
				}
			};
			xhr.send();
		}
	</script>

</template>
