```python
import xml.etree.ElementTree as ET

# Corrected XML data
xml_data = '''
<?xml version="1.0" ?>
<game>
	<title>quest title</title>
Objective: Quest Objective
Description: Quest Description
Rewards: Quest Rewards
QuestGiver: Quest Giver
Tasks:
		<task>
		 Order: Task Order
		 Objective: Task Objective
			Description: Task Description
		 Location: Task Location
		</task>
</game>
'''

# Parse the XML data and fix errors
tree = ET.parse(xml_data)
root = tree.getroot()

# Corrected key completionility errors
for child in root:
    if child.name == 'quest_giver':
        quest_giver = child.text.strip()
        print(f" Quest Giver: {quest_giver}")

# Create a new task with the correct order and objective
new_task = ET.Element("task")
new_task.text = "Order: Task Order"
new_task.appendChild(Et.SubElement(new_task, "objective"))
new_task.appendChild(Et.SubElement(new_task, "description"))

# Add a description to the task
new_task.appendChild(Et.SubElement(new_task, "description"))

# Create a new task with the correct location
new_task = ET.Element("task")
new_task.text = "Location: Task Location"
new_task.appendChild(Et.SubElement(new_task, "location"))

# Add an order to the task
new_task.appendChild(Et.SubElement(new_task, "order"))

# Add a reward to the task
new_task.appendChild(Et.SubElement(new_task, "reward"))
new_task.appendChild(Et.SubElement(new_task, "description"))

# Create a new quest with the correct objectives
newQuest = ET.Element("quest")
newQuest.text = "Quest Objective"
newQuest.appendChild(Et.SubElement(newQuest, "objectives"))
newQuest.appendChild(Et.SubElement(newQuest, "description"))

# Add the task and reward to the quest
newQuest.append(Et.SubElement(newQuest, "task").text)
newQuest.append(Et.SubElement(newQuest, "objective").text)
newQuest.append(Et.SubElement(newQuest, "description").text)
newQuest.append(newQuest, "location").text)
newQuest.append(newQuest, "reward").text)

# Print the updated XML data
print("XML data:")
print(xml_data)
```
