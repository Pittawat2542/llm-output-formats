```python
import xml.etree.ElementTree as ET

# Corrected syntax for XML data
xml_data = '''
<game>
	<id>12345</id>
	<title>The Lost Artifact</title>
	<objective>Retrieve the ancient artifact</objective>
	<description>Your mission is to penetrate the fortress of the enemy, overcome their guards, and retrieve the ancient artifact that has been lost for centuries.</description>
	<reward>5000 gold and a magical sword</reward>
	<quest-giver>The Great Wizard</quest-giver>
	<tasks>
			<task>
				order="2" <objective>Defeat the fortress guards</objective>
				<description>You will face formidable enemies. Prepare for an epic combat.</description>
			的位置>Enchanted Forest</location>
			<task>
				order="3" <objective>Find and retrieve the artifact</ objective>
				<description>The artifact is said to be in a hidden room deep within the fortress. You must find it and return it safely.</description>
			的位置>Fortress Dungeon</location>
			</task>
	</tasks>
</game>
'''

# Parse the XML data
root = ET.fromstring(xml_data)

# Extract the information from the XML data
id = root.find("id")
title = root.find("title")
objective = root.find("objective")
description = root.find("description")
reward = root.find("reward")
quest_giver = root.find("quest-giver")
tasks = [t.find("order") for t in root.findall("task")]

# Output the updated XML data
print("\nUpdated XML data:")
for task in tasks:
    print("- {}, {}".format(task))
```
