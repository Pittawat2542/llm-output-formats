```xml
<?xml version="1.0" ?>
<game>
	<dialogues>
		<dialogue id="1">
			<order>dialogue order</order>
			Speaker name="Speaker1"
			text="Hello, Speaker 1!</text>
		</dialogue>
		<dialogue id="2">
			<order>dialogue order</order>
			 Speaker name="Speaker2"
			text="Hello, Speaker 2!</text>
		</dialogue>
	</dialogues>
</game>
```
