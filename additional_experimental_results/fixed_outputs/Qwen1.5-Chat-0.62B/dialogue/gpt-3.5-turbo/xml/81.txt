def fix_xml(xml_data):
    try:
        # Parse the XML data
        tree = ET.parse(xml_data)

        # Find all elements with the order attribute
        orders = tree.find_all('order')

        # Add the correct order attribute to the speakers

    except Exception as e:
        return f"Error parsing XML: {e}"
