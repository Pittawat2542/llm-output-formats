Conectando os dados da base para a criação de um exemplo de XML formatado com as configurações necessárias para o aplicação em uma lista de tódos exemplos. id, títulos, objetivos, descrições, recompensas e referências, entre outras informações.
```xml
<?xml version="1.0" encoding="UTF-8"?>
<game>
    <id>1</id>
    <title>Quest of the Lost King</title>
    <objective>Find the lost king</objective>
    <description>The king has gone missing and you must find him.</description>
    <reward>100 gold</reward>
    <quest-giver>King of the land</quest-giver>
    <tasks>
        <task>
            <order>1</order>
            <objective>Find the lost king</objective>
            <description>The king has gone missing and you must find him.</description>
            <location>The castle</location>
        </task>
        <task>
            <order>2</order>
            <objective>Find the lost king</objective>
            <description>The king has gone missing and you must find him.</description>
            <location>The castle</location>
        </task>
    </tasks>
</game>
```
