```xml
<?xml version="1.0" ?>
<game>
	<dialogues>
		<dialogue>
			<order>dialogue order</order>
			<speaker>speaker name</speaker>
			<text>dialogue or narration</text>
		</dialogue>
	</dialogues>
</game>
```

# Example:
```xml
<?xml version="1.0" ?>
<game>
	<dialogues>
		<dialogue>
			<order>1</order>
			<speaker>Player</speaker>
			<text>Hello, how are you today?</text>
		</dialogue>
		<dialogue>
			<order>2</order>
			<speaker>NPC</speaker>
			<text>I'm fine, and yourself?</text>
		</dialogue>
	</dialogues>
</game>
```
