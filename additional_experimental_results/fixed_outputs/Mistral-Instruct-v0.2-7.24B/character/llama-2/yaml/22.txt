Also, since you mentioned that you're using `axios`, here's how you could send this request with axios:

##### Axios example:

```javascript
import axios from 'axios';

const data = {
  "$": {
    "game": {
      "character": {
        "$": ["age", 23],
        "$": ["background"],
        "$": ["first_name", "John"],
        "$": ["id", 1],
        "$": ["last_name", "Doe"],
        "$": ["physical_appearance"],
        "$": ["place_of_birth", "New York"],
        "$": ["role", "detective"],
        "$": {"$": ["species.name", "human"]}
      }
    }
  }
};

axios.post('/api/characters', data)
  .then(response => {
    console.log(response.data);
  })
  .catch(error => {
    console.log(error);
  });
```
